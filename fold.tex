% Define a foldable proof command
\newcommand{\foldableproof}[2]{%
  \noindent\textbf{Proof.}~\toggle[ocg={#1}, printocg=true]{[Show/Hide]}\\
  \begin{ocg}{#1}{#1}{0}
  #2
  \end{ocg}
  \qed
}

\section*{Example Theorem}
